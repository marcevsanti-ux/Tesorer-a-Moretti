// ============================================================ // MORETTI S.A. · Tesorería · Sistema de Cheques Rechazados // v1.0.1 — 09/03/2026 // ============================================================ const SHEET_RECHAZADOS = 'RECHAZADOS'; const SHEET_CANCELADOS = 'CANCELADOS'; const SHEET_CONFIG = 'CONFIG'; // Se crea automáticamente // ─── HELPER RESPONSE ───────────────────────────────────────── function respJSON(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ─── GET ───────────────────────────────────────────────────── function doGet(e) { const action = e.parameter && e.parameter.action; if (action === 'getRechazados') return respJSON(getRechazados()); if (action === 'getCancelados') return respJSON(getCancelados()); if (action === 'getConfig') return respJSON(getConfig()); return HtmlService.createHtmlOutputFromFile('Index') .setTitle('Tesorería · Moretti S.A.') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // ─── POST ───────────────────────────────────────────────────── function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; if (action === 'actualizarFila') return respJSON(actualizarFila(body.fila, body.campos)); if (action === 'importarRechazados') return respJSON(importarRechazados(body.rows)); if (action === 'guardarConfig') return respJSON(guardarConfig(body.config)); return respJSON({ ok: false, error: 'Acción no reconocida: ' + action }); } catch(err) { return respJSON({ ok: false, error: err.message }); } } // ─── LEER RECHAZADOS ───────────────────────────────────────── function getRechazados() { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_RECHAZADOS); if (!sheet) return { ok: false, error: 'Hoja RECHAZADOS no encontrada' }; const data = sheet.getDataRange().getValues(); if (data.length < 4) return { ok: true, data: [], kpis: getKPIs([]) }; const rows = []; for (let i = 3; i < data.length; i++) { const r = data[i]; if (!r[0] && !r[5]) continue; rows.push({ fila: i + 1, cliente: String(r[0] || '').trim(), emision: formatDate(r[1]), vencimiento: formatDate(r[2]), tipo: String(r[3] || '').trim(), tipoLetra: String(r[4] || '').trim(), nroCheque: String(r[5] || '').trim(), fechaRechazo: formatDate(r[6]), mes: String(r[7] || '').trim(), anio: String(r[8] || '').trim(), motivo: String(r[9] || '').trim(), importe: parseNum(r[10]), gastos: parseNum(r[11]), total: parseNum(r[12]), detalle: String(r[13] || '').trim(), aviso: formatDate(r[14]), tipoCancelacion: String(r[15] || '').trim(), cancelacion: formatDate(r[16]), importeCancelacion: parseNum(r[17]), pendiente: parseNum(r[18]), envio: String(r[19] || '').trim(), cacDev: String(r[20] || '').trim(), diasMora: parseNum(r[21]), }); } return { ok: true, data: rows, kpis: getKPIs(rows), total: rows.length }; } catch(err) { return { ok: false, error: err.message }; } } // ─── LEER CANCELADOS ───────────────────────────────────────── function getCancelados() { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_CANCELADOS); if (!sheet) return { ok: true, data: [] }; const data = sheet.getDataRange().getValues(); if (data.length < 4) return { ok: true, data: [] }; const rows = []; for (let i = 3; i < data.length; i++) { const r = data[i]; if (!r[0] && !r[5]) continue; rows.push({ fila: i + 1, cliente: String(r[0] || '').trim(), emision: formatDate(r[1]), vencimiento: formatDate(r[2]), tipo: String(r[3] || '').trim(), nroCheque: String(r[5] || '').trim(), fechaRechazo: formatDate(r[6]), mes: String(r[7] || '').trim(), anio: String(r[8] || '').trim(), motivo: String(r[9] || '').trim(), importe: parseNum(r[10]), gastos: parseNum(r[11]), detalle: String(r[13] || '').trim(), cancelacion: formatDate(r[16]), importeCancelacion: parseNum(r[17]), }); } return { ok: true, data: rows, total: rows.length }; } catch(err) { return { ok: false, error: err.message }; } } // ─── ACTUALIZAR FILA ───────────────────────────────────────── function actualizarFila(fila, campos) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_RECHAZADOS); if (!sheet) return { ok: false, error: 'Hoja RECHAZADOS no encontrada' }; const colMap = { detalle: 14, aviso: 15, tipoCancelacion: 16, cancelacion: 17, importeCancelacion: 18, pendiente: 19, envio: 20, cacDev: 21, }; Object.entries(campos).forEach(([key, val]) => { if (colMap[key]) sheet.getRange(fila, colMap[key]).setValue(val); }); return { ok: true, fila }; } catch(err) { return { ok: false, error: err.message }; } } // ─── IMPORTAR RECHAZADOS ────────────────────────────────────── // Recibe array de filas crudas desde el xlsx (SheetJS, header:1) // Columnas esperadas: igual que la planilla (fila 3 = headers, datos desde fila 4) // A=Cliente, B=Emisión, C=Vto, D=TIPO, E=Tipo, F=N°Cheque, // G=FechaRechazo, H=MES, I=AÑO, J=MOTIVO, K=Importe, L=Gastos, // M=Total, N=Detalle, O=Aviso, P=TipoCan, Q=Cancelacion, // R=ImporteCan, S=Pendiente, T=Envio, U=CAC/DEV, V=DiaMora function importarRechazados(rows) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_RECHAZADOS); if (!sheet) return { ok: false, error: 'Hoja RECHAZADOS no encontrada' }; // Encontrar primera fila vacía (después de headers en fila 3) const lastRow = sheet.getLastRow(); let startRow = lastRow + 1; // Verificar que no hay duplicados por N° Cheque const existing = sheet.getRange(4, 6, Math.max(lastRow - 3, 1), 1).getValues().flat().map(String); const duplicados = []; const nuevos = rows.filter(r => { const nro = String(r[5] || '').trim(); if (existing.includes(nro) && nro) { duplicados.push(nro); return false; } return true; }); if (nuevos.length === 0) { return { ok: false, error: 'Todos los cheques ya existen en la planilla. Duplicados: ' + duplicados.join(', ') }; } // Escribir filas nuevas const dataToWrite = nuevos.map(r => [ r[0] || '', // A Cliente r[1] || '', // B Emisión r[2] || '', // C Vencimiento r[3] || '', // D TIPO r[4] || '', // E Tipo letra r[5] || '', // F N° Cheque r[6] || '', // G Fecha Rechazo r[7] || '', // H MES r[8] || '', // I AÑO r[9] || '', // J MOTIVO parseNum(r[10]), // K Importe parseNum(r[11]), // L Gastos parseNum(r[12]), // M Total r[13] || '', // N Detalle r[14] || '', // O Aviso r[15] || '', // P Tipo Cancelación r[16] || '', // Q Cancelación parseNum(r[17]), // R Importe Cancelación parseNum(r[18]), // S Pendiente r[19] || '', // T Envío r[20] || '', // U CAC/DEV parseNum(r[21]), // V Días Mora ]); sheet.getRange(startRow, 1, dataToWrite.length, 22).setValues(dataToWrite); // Notificación por email si está configurado const cfg = getConfig(); if (cfg.mailTo && cfg.onImport) { enviarMailImportacion(cfg, nuevos.length, duplicados.length); } return { ok: true, importados: nuevos.length, duplicados: duplicados.length, filaInicio: startRow, filaFin: startRow + nuevos.length - 1, }; } catch(err) { return { ok: false, error: err.message }; } } // ─── CONFIGURACIÓN ─────────────────────────────────────────── function getConfig() { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_CONFIG); if (!sheet) return defaultConfig(); const data = sheet.getDataRange().getValues(); const cfg = defaultConfig(); data.forEach(row => { if (row[0] && row[1] !== undefined) cfg[row[0]] = row[1]; }); return cfg; } catch(e) { return defaultConfig(); } } function defaultConfig() { return { mailTo: '', mailCC: '', mailSubject: 'Nuevo cheque rechazado — Moretti S.A.', onNew: true, onImport: false, }; } function guardarConfig(config) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_CONFIG); if (!sheet) { sheet = ss.insertSheet(SHEET_CONFIG); sheet.getRange('A1').setValue('clave'); sheet.getRange('B1').setValue('valor'); } // Limpiar y reescribir const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.getRange(2, 1, lastRow - 1, 2).clearContent(); const rows = Object.entries(config).map(([k, v]) => [k, v]); if (rows.length > 0) sheet.getRange(2, 1, rows.length, 2).setValues(rows); return { ok: true }; } catch(err) { return { ok: false, error: err.message }; } } // ─── EMAIL ─────────────────────────────────────────────────── function enviarMailNuevoCheque(row) { try { const cfg = getConfig(); if (!cfg.mailTo || !cfg.onNew) return; const asunto = cfg.mailSubject || 'Nuevo cheque rechazado — Moretti S.A.'; const cuerpo = ` Moretti S.A. — Sistema de Tesorería ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ Se registró un nuevo cheque rechazado: Cliente: ${row.cliente} N° Cheque: ${row.nroCheque} Tipo: ${row.tipo} Fecha Rechazo: ${row.fechaRechazo} Motivo: ${row.motivo} Importe: $${Number(row.importe).toLocaleString('es-AR', {minimumFractionDigits:2})} Gastos: $${Number(row.gastos).toLocaleString('es-AR', {minimumFractionDigits:2})} ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ Este es un mensaje automático del Sistema de Tesorería Moretti S.A. `.trim(); const opts = {}; if (cfg.mailCC) opts.cc = cfg.mailCC; MailApp.sendEmail(cfg.mailTo, asunto, cuerpo, opts); } catch(e) { Logger.log('Error enviando mail: ' + e.message); } } function enviarMailImportacion(cfg, importados, duplicados) { try { const asunto = 'Importación masiva — ' + importados + ' cheques · Moretti S.A.'; const cuerpo = ` Moretti S.A. — Sistema de Tesorería ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ Se realizó una importación masiva de cheques rechazados: Importados: ${importados} registros nuevos Duplicados: ${duplicados} omitidos (ya existían) Fecha: ${Utilities.formatDate(new Date(), 'America/Argentina/Buenos_Aires', 'dd/MM/yyyy HH:mm')} ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ Este es un mensaje automático del Sistema de Tesorería Moretti S.A. `.trim(); const opts = {}; if (cfg.mailCC) opts.cc = cfg.mailCC; MailApp.sendEmail(cfg.mailTo, asunto, cuerpo, opts); } catch(e) { Logger.log('Error enviando mail importación: ' + e.message); } } // ─── KPIs ───────────────────────────────────────────────────── function getKPIs(rows) { const activos = rows.filter(r => r.pendiente > 0); const totalPendiente = activos.reduce((s, r) => s + r.pendiente, 0); const totalImporte = rows.reduce((s, r) => s + r.importe, 0); const totalGastos = rows.reduce((s, r) => s + r.gastos, 0); const moraProm = activos.length ? activos.reduce((s, r) => s + (r.diasMora || 0), 0) / activos.length : 0; const porMotivo = {}; rows.forEach(r => { if (!porMotivo[r.motivo]) porMotivo[r.motivo] = { cantidad: 0, importe: 0 }; porMotivo[r.motivo].cantidad++; porMotivo[r.motivo].importe += r.pendiente; }); const porCliente = {}; rows.forEach(r => { if (!r.cliente) return; if (!porCliente[r.cliente]) porCliente[r.cliente] = { cantidad: 0, pendiente: 0 }; porCliente[r.cliente].cantidad++; porCliente[r.cliente].pendiente += r.pendiente; }); const topClientes = Object.entries(porCliente) .map(([nombre, d]) => ({ nombre, ...d })) .sort((a, b) => b.pendiente - a.pendiente) .slice(0, 8); return { totalPendiente, totalImporte, totalGastos, cantidadActivos: activos.length, cantidadTotal: rows.length, moraProm: Math.round(moraProm), porMotivo, topClientes, }; } // ─── TRIGGER AUTOMÁTICO (configurar manualmente) ───────────── // Para activar el mail automático cuando alguien carga un cheque nuevo: // 1. Ir a Apps Script → Triggers (ícono del reloj) // 2. Agregar trigger: onEdit → hoja RECHAZADOS // 3. Esta función detecta si se editó col A (Cliente) en fila nueva function onEditTrigger(e) { try { const sheet = e.source.getActiveSheet(); if (sheet.getName() !== SHEET_RECHAZADOS) return; if (e.range.getColumn() !== 1) return; // Solo col A = Cliente if (e.range.getRow() < 4) return; // Ignorar headers const row = sheet.getRange(e.range.getRow(), 1, 1, 22).getValues()[0]; if (!row[0] || !row[5]) return; // Necesita cliente y cheque const cfg = getConfig(); if (!cfg.mailTo || !cfg.onNew) return; enviarMailNuevoCheque({ cliente: String(row[0]), nroCheque: String(row[5]), tipo: String(row[3]), fechaRechazo: formatDate(row[6]), motivo: String(row[9]), importe: parseNum(row[10]), gastos: parseNum(row[11]), }); } catch(e) { Logger.log('onEditTrigger error: ' + e.message); } } // ─── HELPERS ───────────────────────────────────────────────── function parseNum(v) { if (v === '' || v === null || v === undefined) return 0; if (typeof v === 'number') return v; return parseFloat(String(v).replace(/[^0-9.-]/g, '')) || 0; } function formatDate(v) { if (!v) return ''; if (v instanceof Date) { return Utilities.formatDate(v, 'America/Argentina/Buenos_Aires', 'dd/MM/yyyy'); } return String(v).trim(); }
